```dataviewjs
let now = dv.date("today");
let start = now.startOf("year");
let end = now.endOf("year");

let pages = dv.pages('"Дневники"')
  .where(p => p.file.name >= start.toFormat("yyyy-MM-dd") 
           && p.file.name <= end.toFormat("yyyy-MM-dd"));

let grouped = pages.groupBy(p => p.type);

dv.table(["Тип", "Количество"],
  grouped.map(g => [g.key, g.rows.length])
);
```
